Correlation Function

cor.index <- function(x,y) {
  nlc <- vector()
  c <- vector()
  Cor.list <- list(nlc, c)
  for (i in 1:length(x)){
    Cor.list$nlc[i] <- c(nlCor(x[[i]],y, method = "log", n=3), nlCor(x[[i]],y, method="poly", n=3), nlCor(x[[i]],y, method = "expo", n=3))
    Cor.list$c[i] <- cor(x[[i]],y)
  }
  return(Cor.list)
}
